     15  mailbox tx2drv1,tx2drv2,tx2sb,mon2sb,tx2cov;
     16  virtual dram_if intf;
     17  configuration cfg;
     18  dram_gen tx;
     19  dram_cov cov;
     20  dram_drv1 drv1;
     21  dram_drv2 drv2;
     22  dram_mon1 mon1;

     25

     28  pkt2=new();
     29  pkt3=new();
     30  pkt4=new();
     31  tx2drv1=new();
     32  tx2drv2=new();
     33  tx2sb=new();
     34  tx2cov=new();
     35  mon2sb=new();

     37  this.cfg=cfg;
     38  
     39  
     40  
     41  tx=new(pkt1,tx2drv1,tx2drv2,tx2sb,tx2cov,cfg);
     42  
     43  cov=new(pkt1,tx2cov,intf);
     44  
     45  drv1=new(pkt1,tx2drv1,intf);
     46  drv2=new(pkt2,tx2drv2,intf);
     47  
     48  mon1=new(pkt3,mon2sb,intf);

     63  drv2.run();

     69  end
