1  `ifndef _Mon1_
     2  `define _Mon1_
     3  `include "pkt.sv"
     4  
     5  class dram_mon1;
     6  dram_pkt pkt;
     7  mailbox mon2sb;
     8  virtual dram_if intf;
     9  extern function new(dram_pkt pkt,mailbox mon2sb,virtual dram_if intf);
     10  extern task run();
     11  endclass
     12  
     13  function dram_mon1::new(dram_pkt pkt,mailbox mon2sb,virtual dram_if intf);
     14  this.pkt=pkt;
     15  this.mon2sb=mon2sb;
     16  this.intf=intf;
     17  endfunction
     18  
     19  task dram_mon1::run();
     20  begin
     21  $display("%t, MON1::RUN PHASE",$time);
     22  while(1)
     23  begin
     24  @(posedge intf.clk);
     25  pkt.en=intf.en;
     26  begin
     27  if((pkt.en==1)&&(intf.wr0==1))
     28  begin
     29  pkt.wr=intf.wr0;
     30  pkt.addr=intf.add0;
     31  pkt.data_out=intf.data0_out;
     32  mon2sb.put(pkt);
     33  $display("%t, monitor1: read addr=%d dataout=%d",$time,pkt.addr,pkt.data_out);
     34  end
     35  end
     36  end
     37  end
     38  endtask
     39  `endif
