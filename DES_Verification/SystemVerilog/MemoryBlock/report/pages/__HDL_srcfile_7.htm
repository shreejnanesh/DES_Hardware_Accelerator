1  `ifndef _drv2_
     2  `define _drv2_
     3  `include "pkt.sv"
     4  
     5  class dram_drv2;
     6  dram_pkt pkt;
     7  mailbox tx2drv2;
     8  virtual dram_if intf;
     9  extern function new(dram_pkt pkt,mailbox tx2drv2,virtual dram_if intf);
     10  extern  task run();
     11  endclass
     12  
     13  function dram_drv2::new(dram_pkt pkt,mailbox tx2drv2,virtual dram_if intf);
     14  this.pkt=pkt;
     15  this.tx2drv2=tx2drv2;
     16  this.intf=intf;
     17  endfunction
     18  
     19  task dram_drv2::run();
     20  $display("%t, DRV2::RUN PHASE",$time);
     21  while(1)
     22  begin
     23  	@(posedge intf.clk);
     24  	tx2drv2.get(pkt);
     25  	pkt.en=1'b1;
     26  	intf.en=pkt.en;
     27  	begin
     28  	if(pkt.wr==1)
     29  	begin
     30  		intf.wr1=pkt.wr;
     31  		intf.add1=pkt.addr;
     32  		$display("%t, driver2:written pkt.wr=%b,pkt.addr=%d",$time,pkt.wr,pkt.addr);
     33  	end
     34  	end
     35  end
     36  endtask
     37  `endif
