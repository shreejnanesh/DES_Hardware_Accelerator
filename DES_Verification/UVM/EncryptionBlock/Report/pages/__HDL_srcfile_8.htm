     7

     12  	  virtual function void build_phase(uvm_phase phase);
     13  	      if (!(uvm_config_db#(virtual enc_interface)::get(this,"","enc_interface", enc_vif))) begin
     14  	         `uvm_fatal("FATAL", "Can't get interface enc_interface\n");

     17  	  endfunction
     18  
     19  	  task run_phase(uvm_phase phase);
     20  	     enc_seq_item itm;
     21  	     while (1) begin
     22  	        itm = enc_seq_item::type_id::create("itm");
     23  	        @(enc_vif.mcb);
     24  	        @(negedge enc_vif.clock);
     25  	        itm.msg = enc_vif.msg;
     26  	        itm.key = enc_vif.key;
     27  	        itm.cipher = enc_vif.cipher;
     28  	        `uvm_info("MON", $sformatf("3: DEBUG: @time %0t, msg=%0d, key=%0d, cipher=%0d", $time, enc_vif.msg, enc_vif.key, enc_vif.cipher), UVM_HIGH);
     29  	        ap.write(itm);
     30  	     end
     31  	  endtask

     33  	endclass
