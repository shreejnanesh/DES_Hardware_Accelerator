1  module DESDEC(input clk,
     2              input [1:64] key, 
     3              input [1:64] cipher,
     4              output reg [1:64] Decrypt ) ;
     5  
     6      reg [1:48] K [1:16];
     7      always@(posedge clk)
     8      begin
     9      	keygen(key,K); 
     10      	Decrypt=decrypt(cipher,K);
     11  
     12      end

     15  task keygen;
     16  input [1:64] key_in;

     26  						2'b10,2'b10,2'b10,2'b10,2'b10,2'b10,2'b01 };
     27  begin
     28  
     29      PC_1={key_in[57],key_in[49],key_in[41],key_in[33],key_in[25],key_in[17],key_in[9]
     30           ,key_in[1],key_in[58],key_in[50],key_in[42],key_in[34],key_in[26],key_in[18]
     31            ,key_in[10],key_in[2],key_in[59],key_in[51],key_in[43],key_in[35],key_in[27]

     36              ,key_in[21],key_in[13],key_in[5],key_in[28],key_in[20],key_in[12],key_in[4]};
     37  
     38      C0=PC_1[1:28];
     39      D0=PC_1[29:56];
     40  
     41  for(i=1;i<=16;i++)
     42  begin
     43  	C1=rotate_left(C0,lf[i]);

     63  function [1:28] rotate_left([1:28] temp,[1:2] lf);
     64  reg [1:28] out;
     65  begin
     66      out = ((temp << lf)|(temp >> ($bits(temp)-lf)));
     67      return out;
     68  end

     100  function [1:64] fiestal_round([1:64] in, [1:48] key);

     102  reg [1:32] l0;
     103  reg [1:32] r0;
     104  reg [1:48] Ebit;
     105  reg [1:48] f1;
     106  reg [1:32] S;
     107  reg [1:32] P;
     108  reg [1:32] l1;
     109  reg [1:32] r1;
     110  begin
     111      r0 = in[33:64] ;
     112      l0 = in[1:32];
     113      Ebit={r0[32],   r0[1]  ,  r0[2]  ,   r0[3] ,    r0[4]  ,  r0[5],
     114      	r0[4]  ,   r0[5]  ,  r0[6]   ,  r0[7]   ,  r0[8]  ,  r0[9],

     117      	r0[16] ,  r0[17]  , r0[18]  ,  r0[19]  ,  r0[20] ,  r0[21],
     118      	r0[20]  ,  r0[21]  , r0[22]  ,  r0[23] ,  r0[24] ,  r0[25],
     119      	r0[24]  ,  r0[25] , r0[26]  ,  r0[27]  ,  r0[28]  , r0[29],
     120      	r0[28]   , r0[29]  , r0[30] ,   r0[31]  ,  r0[32]  ,  r0[1]};
     121  
     122      f1=key^Ebit;
     123  
     124      S={sbox1(f1[1:6]),sbox2(f1[7:12]),sbox3(f1[13:18]),sbox4(f1[19:24]),sbox5(f1[25:30]),
     125      	sbox6(f1[31:36]),sbox7(f1[37:42]),sbox8(f1[43:48])};
     126  
     127      P = {S[16], S[7], S[20], S[21], S[29], S[12], S[28],
     128      			S[17], S[1], S[15], S[23], S[26], S[5], S[18],
     129      			S[31], S[10], S[2], S[8], S[24], S[14], S[32],
     130      			S[27], S[3], S[9], S[19], S[13], S[30], S[6],
     131      			S[22], S[11], S[4], S[25]};
     132      r1=l0^P;
     133      l1=r0;
     134      out={l1[1:32],r1[1:32]};
     135      return out;
     136  end
     137  endfunction
