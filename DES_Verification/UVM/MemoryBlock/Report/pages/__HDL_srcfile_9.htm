1  class dram_mon1 extends uvm_monitor;
     2  `uvm_component_utils(dram_mon1)
     3  dram_seq_item pkt;
     4  virtual intif inf;
     5  
     6  uvm_analysis_port #(dram_seq_item) item_collected_port;
     7  uvm_analysis_port #(dram_seq_item) custom_ap;
     8  
     9  function new(string name="dram_mon1",uvm_component parent);
     10  super.new(name,parent);
     11  endfunction
     12  
     13  virtual function void build_phase(uvm_phase phase);
     14  super.build_phase(phase);
     15  uvm_config_db #(virtual intif) ::get(this,"","inf",inf);
     16  item_collected_port=new("item_collected_port",this);
     17  custom_ap = new("analysis_port",this);
     18  endfunction
     19  
     20  task run_phase(uvm_phase phase);
     21  pkt=dram_seq_item::type_id::create("pkt");
     22  forever
     23  begin
     24  #5;
     25  @(posedge inf.clk)
     26  if((inf.en==1)&&(inf.wr0==0)) begin
     27  pkt.en=inf.en;
     28  pkt.wr=inf.wr0;
     29  pkt.data_in=inf.data0_in;
     30  pkt.add=inf.add0;	
     31  `uvm_info("MON1","MON1 TRANSACTIONS" ,UVM_NONE);
     32  end
     33  `uvm_info("MON","MON TRANSACTIONS",UVM_NONE);
     34  @(posedge inf.clk)
     35  item_collected_port.write(pkt);
     36  custom_ap.write (pkt);
     37  	end
     38  endtask
     39  endclass
