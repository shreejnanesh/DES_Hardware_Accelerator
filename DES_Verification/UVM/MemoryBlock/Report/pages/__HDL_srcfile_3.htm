     5  module DUT(intif.dut inf);
     6  logic[63:0] mem[0:63];
     7  always@(posedge inf.clk)
     8  begin
     9  if(inf.en == 0) begin
     10  	inf.data0_out <= 8'd0;
     11  	inf.data1_out <= 8'd0;
     12  end
     13  else if(inf.wr0==0)
     14  	mem[inf.add0] <= inf.data0_in;
     15  else
     16  	inf.data0_out <= mem[inf.add0];
     17  end
     18  
     19  always@(posedge inf.clk)
     20  begin
     21  if(inf.en == 0) begin
     22  	inf.data0_out <= 8'd0;
     23  	inf.data1_out <= 8'd0;
     24  end
     25  else if(inf.wr1==0)
     26  	mem[inf.add1] <= inf.data1_in;
     27  else
     28  	inf.data1_out <= mem[inf.add1];
     29  end
     30  
     31  endmodule
